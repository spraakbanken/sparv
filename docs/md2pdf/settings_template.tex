\usepackage[utf8]{inputenc}
\usepackage[T1]{fontenc} % Use 8-bit encoding that has 256 glyphs
\usepackage{fourier} % Use the Adobe Utopia font for the document
\usepackage[english]{babel} % English language/hyphenation

\usepackage[scaled]{helvet} % Use Helvetica font
\renewcommand\familydefault{\sfdefault}

\usepackage{graphicx} % For including graphics
\usepackage[dvipsnames]{xcolor} % Display colors
\usepackage{pmboxdraw} % Display funny tree structure chars

% Highlight inline code
\usepackage{xcolor}
\definecolor{codebg}{HTML}{EEEEEE}
\newcommand{\inlinecode}[1]{\mbox{\setlength{\fboxsep}{1pt}\colorbox{codebg}{\texttt{#1}}}}

\usepackage{sectsty} % Allows customizing section commands
\allsectionsfont{\normalfont\bf} % Make all sections centered, the default font and bold

\usepackage[a4paper, margin=3cm]{geometry} % Smaller margins
% \linespread{1.1} % Line spacing

\usepackage{titlesec} % Allows customizing section commands

\usepackage{array} % For custom column widths in tables and increasing row height in tables
\renewcommand{\arraystretch}{1.6} % Increase row height in tables
\usepackage{calc} % for calculating minipage widths

% Remove the "Chapter" prefix from chapter titles
\usepackage{titlesec}
\titleformat{\chapter}[display]
  {\normalfont\huge\bfseries}{}{0pt}{\Huge}

% Fix numbering of chapters and sections
\renewcommand{\thechapter}{}
\renewcommand{\thesection}{\arabic{section}}
\setcounter{section}{0} % Reset the section counter
\renewcommand{\thesubsection}{\arabic{section}.\arabic{subsection}}

\titleformat{\paragraph}[hang]{\normalfont\bfseries}{\theparagraph}{1em}{}
\titlespacing*{\paragraph}{0pt}{3.25ex plus 1ex minus .2ex}{0.5em} % Line break after paragraph

\titleformat{\subparagraph}[hang]{\normalfont\bfseries}{\thesubparagraph}{1em}{}
\titlespacing*{\subparagraph}{0pt}{3.25ex plus 1ex minus .2ex}{0.5em} % Line break after paragraph

\usepackage{fancyhdr} % Custom headers and footers
\pagestyle{fancyplain} % Makes all pages in the document conform to the custom headers and footers
\fancyhead{} % No page header
\fancyfoot[L]{} % Empty left footer
\fancyfoot[C]{} % Empty center footer
\fancyfoot[R]{\thepage} % Page numbering for right footer
\renewcommand{\headrulewidth}{0pt} % Remove header underlines
\renewcommand{\footrulewidth}{0pt} % Remove footer underlines
\setlength{\headheight}{13.6pt} % Customize the height of the header

\setlength\parindent{0pt} % Removes all indentation from paragraphs

\widowpenalty10000 % Prevent widows
\clubpenalty10000 % Prevent orphans

% Display of code blocks
\lstset{
    basicstyle=\ttfamily,
    showspaces=false,
    showstringspaces=false, % Remove unnecessary space markers
    showtabs=false,
    tabsize=2,
    captionpos=b,
    breaklines=true,
    breakatwhitespace=false,
    breakautoindent=true,
    linewidth=\textwidth,
    inputencoding=utf8,
    extendedchars=true,
    literate=

        {└}{\textSFii}1 {├}{\textSFviii}1 {─}{\textSFx}1 {│}{\textSFxi}1,
    keywordstyle=\color[rgb]{0.13,0.29,0.53}\bfseries,
    stringstyle=\color[rgb]{0.31,0.60,0.02},

    basewidth=0.5em,
}

% Uniform bullet points on all levels
\usepackage{enumitem}
\setlist[itemize]{label=\textbullet}

% Prevent page breaks for listings
\usepackage{etoolbox}
\BeforeBeginEnvironment{lstlisting}{\par\noindent\begin{minipage}{\linewidth}}
\AfterEndEnvironment{lstlisting}{\end{minipage}\par\addvspace{\topskip}}

% New page for new chapter
\newcommand{\sectionbreak}{\clearpage}

% Prettier block quotes
\usepackage[most]{tcolorbox}
\let\oldquote=\quote
\let\endoldquote=\endquote
\definecolor{bkg-color}{gray}{0.95}
\definecolor{mark-color}{HTML}{17a2b8}
\newtcolorbox{myblockquote}{
    breakable, % IMPORTANT
    left=0pt,right=0pt,
    colback=bkg-color,
    boxrule=0pt,
    arc=0pt,
    boxsep=0pt,
    enhanced jigsaw,
    borderline west={4pt}{0pt}{mark-color},
}
\renewenvironment{quote}{\begin{myblockquote}\begin{oldquote}}{\end{oldquote}\end{myblockquote}}
